# Полный формат с логином и паролем
socks5://login:password@host:port
socks5://host:port:login:password 
protocol:host:port:login:password

# Формат только с паролем
socks5://host:port::password
protocol:host:port:password

# Простой формат без авторизации
socks5://host:port
protocol:host:port

# Примеры:
socks5://proxy.example.com:1080:user:pass
socks5://proxy.example.com:1080:pass
socks5://user:pass@proxy.example.com:1080
http://proxy.example.com:8080
proxy.example.com:1080:user:pass  # будет считаться как socks5
